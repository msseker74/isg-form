function konularTemizleVeYukle() {
  confirm2("Tüm kayıtlı konular silinip CSV'den yeniden yüklenecek. Kitaplar etkilenmez. Devam?", function() {
    // 1. Yerel KAYNAKLAR.dersler'i temizle

    // 2. Sheets'teki konu satırlarını temizle, sonra CSV yükle
    if (!DEMO_MODE && typeof google !== "undefined" && google.script) {
      syncStart();
      google.script.run
        .withSuccessHandler(function(result) {
          syncEnd();
          if (result && result.success) {
            toast("✅ Konular silindi. Şimdi CSV seçin...");
            setTimeout(() => konuListesiYukle(), 600);
          } else {
            toast("❌ " + ((result && result.error) || "Silme hatası"), true);
          }
        })
        .withFailureHandler(function(e) {
          syncEnd();
          toast("❌ Bağlantı hatası: " + e.message, true);
        })
        .konulariTemizle();
    } else {
      // Demo modda sadece yerel temizle ve CSV yükle
      toast("✅ Demo mod: Konular temizlendi. Şimdi CSV seçin...");
      setTimeout(() => konuListesiYukle(), 600);
    }
  });
}

function konuListesiYukle() {
  const input = document.createElement("input");
  input.type = "file";
  input.accept = ".csv";
  input.onchange = function(e) {
    const file = e.target.files[0];
    if (!file) return;
    
    const reader = new FileReader();
    reader.onload = function(ev) {
      const text = ev.target.result.replace(/^\uFEFF/, "");
      const lines = text.split("\n").filter(l => l.trim() !== "");
      
      if (lines.length < 2) {
        alert("CSV dosyası çok kısa!");
        return;
      }
      
      // Başlık satırını al (dersler)
      const headers = lines[0].split(";").map(h => h.trim());
      
      // Ders listesi (Türkçe karakter düzeltmesi)
      const dersler = headers.map(h => {
        if (h.includes("Matematik")) return "Matematik";
        if (h.includes("Fizik")) return "Fizik";
        if (h.includes("Kimya")) return "Kimya";
        if (h.includes("Biyoloji")) return "Biyoloji";
        if (h.includes("Türkçe")) return "Türkçe";
        if (h.includes("Tarih")) return "Tarih";
        if (h.includes("Coğrafya")) return "Coğrafya";
        if (h.includes("İngilizce")) return "İngilizce";
        if (h.includes("Felsefe")) return "Felsefe";
        return h;
      });
      
      // Konuları topla (ünite başlıkları dahil)
      const konular = {};
      dersler.forEach(d => { konular[d] = []; });

      // Her satırı işle (başlık satırı atla, i=1'den başla)
      for (let i = 1; i < lines.length; i++) {
        const cols = lines[i].split(";");   // trim YAPMA — girinti bilgisi önemli

        for (let j = 0; j < cols.length; j++) {
          const ders = dersler[j];
          if (!ders) continue;

          const ham = cols[j];                    // ham hali (girintili olabilir)
          if (!ham || ham.trim() === "") continue;

          // ── ATLANACAK SATIRLAR ───────────────────────────────────────
          // "Alıştırmalar", "Ön Değerlendirme", "Tema Ölçme...", boş/kısa satırlar
          const trimmed = ham.trim();
          const atla =
            /^Alıştırmalar$/i.test(trimmed) ||
            /^Ön Değerlendirme$/i.test(trimmed) ||
            /Ölçme ve Değerlendirme/i.test(trimmed) ||
            /^Tema\s*:/i.test(trimmed) ||           // "Tema: GEOMETRİK ŞEKİLLER"
            /^\d+\.\s*Tema\b/i.test(trimmed) ||   // "1. Tema: ..."  veya "2. Tema ..."
            /^\d+\.\s*$/.test(trimmed) ||           // sadece "1." gibi
            trimmed.length <= 2;

          if (atla) continue;

          // ── İÇERİĞİ TEMİZLE ─────────────────────────────────────────
          let temiz = trimmed
            .replace(/^-\s*/, "")   // baştaki tire kaldır
            .replace(/\s+/g, " ")
            .trim();

          if (temiz.length <= 2) continue;

          // Tekrar kontrolü ile ekle (sıra korunur, alfabetik sıralama YOK)
          if (!konular[ders].includes(temiz)) {
            konular[ders].push(temiz);
          }
        }
      }

      // Benzersizlik garantisi (sıra korunur)
      Object.keys(konular).forEach(ders => {
        konular[ders] = [...new Set(konular[ders])];
      });
      
      // Özet bilgi hazırla
      let ozetMesaj = "📊 İŞLENEN KONULAR:\n\n";
      let toplamKonu = 0;
      
      Object.keys(konular).forEach(ders => {
        if (konular[ders].length > 0) {
          ozetMesaj += `${ders}: ${konular[ders].length} konu\n`;
          toplamKonu += konular[ders].length;
        }
      });
      
      ozetMesaj += `\nToplam ${toplamKonu} konu işlendi.`;
      console.log(ozetMesaj);
      
      // ÖNCE Google Sheets'e kaydet (backend)
      if (!DEMO_MODE && typeof google !== "undefined" && google.script) {
        syncStart();
        
        google.script.run
          .withSuccessHandler(function(result) {
            syncEnd();
            
            if (result.success) {
              // Başarılı olursa, yerel KAYNAKLAR'ı da güncelle
              if (!KAYNAKLAR.dersler) KAYNAKLAR.dersler = {};
              
              Object.keys(konular).forEach(ders => {
                if (!KAYNAKLAR.dersler[ders]) KAYNAKLAR.dersler[ders] = [];
                konular[ders].forEach(konu => {
                  if (!KAYNAKLAR.dersler[ders].includes(konu)) {
                    KAYNAKLAR.dersler[ders].push(konu);
                  }
                });
                KAYNAKLAR.dersler[ders].sort();
              });
              
              // Detaylı toast mesajı
              let detayMesaj = `✅ ${result.eklenen} yeni konu eklendi`;
              if (result.guncellenen > 0) {
                detayMesaj += `, ${result.guncellenen} konu zaten mevcuttu`;
              }
              toast(detayMesaj);
              
              // Konsola detaylı bilgi
              console.log(`📋 Google Sheets sonucu: ${result.eklenen} eklenen, ${result.guncellenen} mevcut`);
              
              // Kaynaklar sekmesini yenile
              setTab("kaynaklar");
            } else {
              toast("❌ " + (result.error || "Kaydetme hatası"), true);
            }
          })
          .withFailureHandler(function(e) {
            syncEnd();
            toast("❌ Bağlantı hatası: " + e.message, true);
            
            // Hata durumunda yerel KAYNAKLAR'a ekle (yedek)
            if (!KAYNAKLAR.dersler) KAYNAKLAR.dersler = {};
            
            let eklenenToplam = 0;
            Object.keys(konular).forEach(ders => {
              if (!KAYNAKLAR.dersler[ders]) KAYNAKLAR.dersler[ders] = [];
              konular[ders].forEach(konu => {
                if (!KAYNAKLAR.dersler[ders].includes(konu)) {
                  KAYNAKLAR.dersler[ders].push(konu);
                  eklenenToplam++;
                }
              });
            });
            
            toast(`⚠️ Bağlantı hatası, ${eklenenToplam} konu yerel olarak kaydedildi`, true);
            setTab("kaynaklar");
          })
          .topluKonuEkle(konular); // Backend'deki topluKonuEkle fonksiyonunu çağır
          
      } else {
        // Demo modunda veya bağlantı yoksa yerel KAYNAKLAR'a ekle
        if (!KAYNAKLAR.dersler) KAYNAKLAR.dersler = {};
        
        let eklenenToplam = 0;
        Object.keys(konular).forEach(ders => {
          if (!KAYNAKLAR.dersler[ders]) KAYNAKLAR.dersler[ders] = [];
          konular[ders].forEach(konu => {
            if (!KAYNAKLAR.dersler[ders].includes(konu)) {
              KAYNAKLAR.dersler[ders].push(konu);
              eklenenToplam++;
            }
          });
          KAYNAKLAR.dersler[ders].sort();
        });
        
        toast(`✅ Demo mod: ${eklenenToplam} konu yerel olarak eklendi`);
        setTab("kaynaklar");
      }
    };
    reader.readAsText(file, "UTF-8");
  };
  input.click();
}

function parseOdev(a) {
  return {
    ...a,
    verilen: numFmt(a.verilen),
    cozulen: numFmt(a.cozulen),
    dogru: numFmt(a.dogru),
    yanlis: numFmt(a.yanlis),
    bos: numFmt(a.bos),
    net: typeof a.net === 'string' ? parseFloat(a.net.replace(',', '.')) : numFmt(a.net),
    kocPuan: a.kocPuan !== "" && a.kocPuan !== null ? numFmt(a.kocPuan) : null,
    durum: a.durum || "pending"
  };
}

function doLogin() {
  const u = document.getElementById("loginUser").value.trim().toLowerCase();
  const p = document.getElementById("loginPass").value;
  let user = null;
  
  if (loginRole === "koc") {
    if (u === KOC_KULLANICI.toLowerCase() && p === KOC_SIFRE) {
      user = { ...COACH_DATA, role: "koc" };
    }
  } else {
    user = DB.ogrenciler.find(o =>
      o.kullaniciAdi && o.kullaniciAdi.trim().toLowerCase() === u &&
      o.sifre && String(o.sifre) === String(p)
    );
    if (user) user = { ...user, role: "ogr" };
  }
  
  if (!user) {
    document.getElementById("loginErr").style.display = "block";
    return;
  }
  
  document.getElementById("loginErr").style.display = "none";
  SESSION = user;
  sessionStorage.setItem("ec_session", JSON.stringify(user));
  showApp();
}

function yenile() {
  if (DEMO_MODE) {
    toast("✅ Demo modda yenileme yok");
    return;
  }
  if (typeof google === "undefined") return;

    .withSuccessHandler(function(r) {

      DB.odevler = DB.odevler.map(parseOdev);
      if (r.ayarlar && r.ayarlar.ad) {
        COACH_DATA = { ...COACH_DATA, 
          ad: r.ayarlar.ad,
          kullaniciAdi: r.ayarlar.kullaniciAdi || COACH_DATA.kullaniciAdi,
          sifre: r.ayarlar.sifre || COACH_DATA.sifre
        };
      }
      if (r.sablonlar && r.sablonlar.length > 0) {
        SABLONLAR = r.sablonlar.map(s => ({ ...s, verilen: Number(s.verilen) || 20 }));
      }
      if (r.kaynaklar) KAYNAKLAR = r.kaynaklar;
      if (r.sinavlar) DB.sinavlar = r.sinavlar;
      
      if (SESSION.role === "ogr") {
        const güncelOgr = DB.ogrenciler.find(o => o.id === SESSION.id);
        if (güncelOgr) SESSION = { ...güncelOgr, role: "ogr" };
      }
      
      if (SESSION.role === "koc") setTab(aktifTab || "dashboard");
      else setTabOgr(aktifTabOgr || "odevlerim");
      
      toast("✅ Veriler güncellendi");
    })
    .withFailureHandler(e => toast("❌ " + e.message, true))
    .getData();
}

function verileriSenkronizeEt(showToastMsg = true) {
  // Sadece koç senkronize edebilir — öğrenci girişinde veri ezilmesin
  if (SESSION && SESSION.role !== "koc") return;
  if (DEMO_MODE) {
    if (showToastMsg) toast("✅ Demo modda senkronizasyon gerekmez");
    return;
  }
  
  if (typeof google === "undefined" || !google.script) {
    if (showToastMsg) toast("❌ Google Apps Script bağlantısı yok", true);
    return;
  }
  
  syncStart();
  
  google.script.run
    .withSuccessHandler(function(result) {
      syncEnd();
      if (result.success) {
        toast("✅ " + result.mesaj);
        yenile(); // Verileri Sheets'ten tekrar çek ve ekranı tazele
      } else {
        toast("❌ " + result.error, true);
      }
    })
    .withFailureHandler(function(e) {
      syncEnd();
      toast("❌ Kayıt başarısız: " + e.message, true); 
    })
    .topluGuncelle({
      ogrenciler: DB.ogrenciler,
      odevler: DB.odevler,
      hedefler: DB.hedefler,
      sinavlar: DB.sinavlar || [],
      kaynaklar: KAYNAKLAR,
      sablonlar: SABLONLAR
    });
}

}

function haftaBasi() {
  const now = new Date();
  const d = now.getDay();
  const m = new Date(now);
  m.setDate(now.getDate() - (d === 0 ? 6 : d - 1));
  return m.toISOString().slice(0, 10);
}

function haftaGunleri(offset) {
  const now = new Date();
  const d = now.getDay();
  const mon = new Date(now);
  mon.setDate(now.getDate() - (d === 0 ? 6 : d - 1) + (offset || 0) * 7);
  return Array.from({ length: 7 }, (_, i) => {
    const x = new Date(mon);
    x.setDate(mon.getDate() + i);
    return x.toISOString().slice(0, 10);
  });
}

function gecmis(tarih) {

function streak(odevler) {
  const gunler = new Set(odevler.filter(a => a.durum !== "pending").map(a => fmt(a.tarih)));
  let s = 0, d = new Date();
  while (gunler.has(d.toISOString().slice(0, 10))) {
    s++;
    d.setDate(d.getDate() - 1);
  }
  return s;
}

function trendHesapla(odevler, period) {
  const now = new Date();
  let haftalar;
  
  if (period === "gunluk") {
    haftalar = Array.from({ length: 7 }, (_, i) => {
      const d = new Date(now);
      d.setDate(now.getDate() - (6 - i));
      const ds = d.toISOString().slice(0, 10);
      const items = odevler.filter(a => fmt(a.tarih) === ds && a.cozulen > 0);
      return items.reduce((s, a) => s + (a.dogru - a.yanlis * 0.25), 0);
    });
  } else if (period === "haftalik") {
    haftalar = Array.from({ length: 8 }, (_, i) => {
      const from = new Date(now);
      from.setDate(now.getDate() - (7 - i) * 7 - 6);
      const to = new Date(now);
      to.setDate(now.getDate() - (7 - i) * 7);
      const items = odevler.filter(a => {
        const d = new Date(fmt(a.tarih));
        return d >= from && d <= to && a.cozulen > 0;
      });
      return items.reduce((s, a) => s + (a.dogru - a.yanlis * 0.25), 0);
    });
  } else if (period === "aylik") {
    haftalar = Array.from({ length: 12 }, (_, i) => {
      const d = new Date(now.getFullYear(), now.getMonth() - 11 + i, 1);
      const dSon = new Date(now.getFullYear(), now.getMonth() - 11 + i + 1, 0);
      const items = odevler.filter(a => {
        const t = new Date(fmt(a.tarih));
        return t >= d && t <= dSon && a.cozulen > 0;
      });
      return items.reduce((s, a) => s + (a.dogru - a.yanlis * 0.25), 0);
    });
  } else {
    haftalar = Array.from({ length: 4 }, (_, i) => {
      const y = now.getFullYear() - 3 + i;
      const items = odevler.filter(a => {
        const t = new Date(fmt(a.tarih));
        return t.getFullYear() === y && a.cozulen > 0;
      });
      return items.reduce((s, a) => s + (a.dogru - a.yanlis * 0.25), 0);
    });
  }
  
  const son = haftalar[haftalar.length - 1];
  const onceki = haftalar[haftalar.length - 2] || 0;
  const fark = son - onceki;
  
  if (Math.abs(fark) < 0.5) return { icon: "→", cls: "trend-same", text: "Stabil", fark: 0, veri: haftalar };
  if (fark > 0) return { icon: "↑", cls: "trend-up", text: `+${fark.toFixed(1)}`, fark, veri: haftalar };
  return { icon: "↓", cls: "trend-down", text: fark.toFixed(1), fark, veri: haftalar };
}

        const t = new Date(fmt(a.tarih));

      });
      return {
        label: String(y),
        net: parseFloat(items.reduce((s, a) => s + (a.dogru - a.yanlis * 0.25), 0).toFixed(1))
      };

function renderOdevAnalizIcerik(el, ogrId) {
  const odevler = DB.odevler.filter(a => !ogrId || a.studentId === ogrId);
  
  if (odevler.length === 0) {
    el.innerHTML = `<div class="card" style="text-align:center; padding:40px; color:#475569">Henüz ödev verisi bulunamadı.</div>`;
    return;
  }
  
  const totCoz = odevler.reduce((s, a) => s + (a.cozulen || 0), 0);
  const totD = odevler.reduce((s, a) => s + (a.dogru || 0), 0);
  const totY = odevler.reduce((s, a) => s + (a.yanlis || 0), 0);
  const totNet = (totD - totY * 0.25).toFixed(2);
  const tr = trendHesapla(odevler, analizPeriod);
  
  const subMap = {};
  const konuMap = {};
  odevler.forEach(a => {
    if (!subMap[a.ders]) subMap[a.ders] = { cozulen: 0, dogru: 0, yanlis: 0, count: 0 };
    subMap[a.ders].cozulen += a.cozulen || 0;
    subMap[a.ders].dogru += a.dogru || 0;
    subMap[a.ders].yanlis += a.yanlis || 0;
    subMap[a.ders].count++;

function renderOgr(tab) {
  const el = document.getElementById("tab-" + tab);
  if (!el) return;
  const myId = SESSION.id;
  const odevlerim = DB.odevler.filter(a => a.studentId === myId);
  const hedeflerim = DB.hedefler.filter(h => h.studentId === myId);
  const wb = haftaBasi();

  const doRender = () => {
    if (tab === "odevlerim") renderOgrOdevler(el, odevlerim, hedeflerim, wb);
    else if (tab === "haftalikplanim") renderOgrHaftalikPlan(el, myId);
    else if (tab === "analizim") renderOgrAnaliz(el, odevlerim);
    else if (tab === "hedeflerim") renderOgrHedefler(el, odevlerim, hedeflerim, wb);
    else if (tab === "yorumlar") renderOgrYorumlar(el, odevlerim);
    else if (tab === "rozetlerim") renderOgrRozetler(el, odevlerim);
  };

  const AGIR_OGR = new Set(["analizim", "haftalikplanim"]);
  if (AGIR_OGR.has(tab)) {
    el.innerHTML = skeletonHTML(3);
    requestAnimationFrame(() => setTimeout(doRender, 50));
  } else {
    doRender();
  }
}

// Akordiyon açma/kapama fonksiyonu 
function toggleDersAnaliz(index) {
  const content = document.getElementById(`ders-analiz-content-${index}`);
  const icon = document.getElementById(`ders-analiz-icon-${index}`);
  
  if (content.style.display === 'none' || content.style.display === '') {
    content.style.display = 'block';
    icon.textContent = '▼';
    icon.style.color = '#a78bfa';
  } else {
    content.style.display = 'none';
    icon.textContent = '▶';
    icon.style.color = ''; // eski rengine döner
  }
}

// ✨ GÜNCELLENDİ: Mükerrer kontrol ve başarılı kayıt mesajı eklendi
function odevKaydet(id) {
  const konu = document.getElementById("mKonu").value.trim();
  if (!konu) { alert("Konu zorunlu!"); return; }
  const mevcut = id ? DB.odevler.find(a => a.id === id) : null;

  const data = {
    id: id || uid(),
    studentId: document.getElementById("mOgr").value,
    ders: document.getElementById("mDers").value,
    konu: konu,
    tarih: document.getElementById("mTarih").value,
    odevTipi: (document.getElementById("secilenOdevTipi") ? document.getElementById("secilenOdevTipi").value : (mevcut ? (mevcut.odevTipi || "test") : "test")),
    verilen: Number(document.getElementById("mVerilen") ? document.getElementById("mVerilen").value : 0) || 0,
    cozulen: mevcut ? mevcut.cozulen : 0,
    dogru: mevcut ? mevcut.dogru : 0,
    yanlis: mevcut ? mevcut.yanlis : 0,
    bos: mevcut ? mevcut.bos : 0,
    net: mevcut ? mevcut.net : 0,
    durum: mevcut ? mevcut.durum : "pending",
    not: document.getElementById("mYonerge") ? document.getElementById("mYonerge").value.trim() : (mevcut ? mevcut.not : ""),
    kocNot: document.getElementById("mKocNot") ? document.getElementById("mKocNot").value.trim() : (mevcut ? mevcut.kocNot : ""),
    kocPuan: mevcut ? mevcut.kocPuan : null,
    olusturma: mevcut ? mevcut.olusturma : bugun(),
    kaynakKitap: document.getElementById("mKitap").value || "",
    sayfaNo: document.getElementById("mSayfaNo").value.trim() || ""
  };

  // Mükerrer kontrol (sadece yeni kayıtlarda)
  if (!id) {
    const ayniOgrAyniKonu = DB.odevler.find(o => 
      o.studentId === data.studentId && 
      o.ders === data.ders && 
      o.konu === data.konu && 
      o.tarih === data.tarih
    );
    
    if (ayniOgrAyniKonu) {
      toast("⚠️ Bu öğrenci için aynı ders/konu/tarihte ödev zaten mevcut!", true);
      return;
    }
  }

  if (id) {
    DB.odevler = DB.odevler.map(o => o.id === id ? parseOdev(data) : o);
  } else {
    DB.odevler.push(parseOdev(data));
  }

  if (!DEMO_MODE && typeof google !== "undefined") {
    google.script.run
      .withSuccessHandler((result) => {
        if (result.success) {
          toast("✅ " + (result.mesaj || "Ödev kaydedildi!"));
        } else {
          toast("❌ " + (result.error || "Kayıt hatası"), true);
        }
      })
      .withFailureHandler(() => toast("❌ Bağlantı hatası", true))
      .hizliOdevKaydet(data);
  } else {
    toast("✅ Ödev başarıyla kaydedildi!");
  }

  setTab(aktifTab === "plan" ? "plan" : "odevler");
  modalKapat();
}

// ✨ GÜNCELLENDİ: Silme işlemi mesajı eklendi
function odevSil(id) {
  if (SESSION && SESSION.role === "ogr") {
    toast("⚠️ Yetkisiz İşlem: Öğrenciler ödev silemez!", true);
    return;
  }

  confirm2("Bu ödevi silmek istediğine emin misin?", function() {
    // 1. Ekrandan sil
    DB.odevler = DB.odevler.filter(a => a.id !== id);
    filterOdevler();
    toast("🗑 Ödev silindi");
    
    // 2. Sheets'ten sil
    if (!DEMO_MODE) {
      google.script.run
        .withSuccessHandler((result) => {
          if (!result.success) toast("Hata: " + result.error, true);
        })
        .withFailureHandler(e => toast("Hata: Silme yansımadı!", true))
        .hizliSatirSil("Odevler", id);
    }
  });
}

// ✨ GÜNCELLENDİ: Mükerrer kontrol ve başarılı kayıt mesajı eklendi
function ogrenciKaydet(id) {
  const ad = document.getElementById("oAd").value.trim();
  const kadi = document.getElementById("oKadi").value.trim().toLowerCase();
  if (!ad || !kadi) { alert("Ad ve kullanıcı adı zorunlu!"); return; }
  
  // Mükerrer kullanıcı adı kontrolü
  const dupCheck = DB.ogrenciler.find(o => o.kullaniciAdi === kadi && o.id !== id);
  if (dupCheck) { 
    toast("❌ Bu kullanıcı adı zaten kullanılıyor!", true);
    return; 
  }
  
  const data = {
    id: id || uid(),
    ad,
    kullaniciAdi: kadi,
    sifre: document.getElementById("oSifre").value || "1234",
    sinif: document.getElementById("oSinif").value,
    renk: document.getElementById("oRenk").value,
    kocId: "coach1"
  };
  
  if (id) {
    DB.ogrenciler = DB.ogrenciler.map(o => o.id === id ? data : o);
  } else {
    DB.ogrenciler.push(data);
  }

  if (!DEMO_MODE) {
    google.script.run
      .withSuccessHandler((result) => {
        if (result.success) {
          toast("✅ " + (result.mesaj || "Öğrenci kaydedildi!"));
        } else {
          toast("❌ " + (result.error || "Kayıt hatası"), true);
        }
      })
      .withFailureHandler(() => toast("❌ Bağlantı hatası", true))
      .hizliOgrenciKaydet(data);
  } else {
    toast("✅ Öğrenci başarıyla kaydedildi!");
  }

  modalKapat();
  setTab("ogrenciler");
}

function ogrOdevTeslimEt(id) {
  const a = DB.odevler.find(x => x.id === id);
  if (!a) return;
  const isEtkinlik = a.odevTipi === "etkinlik";
  const not = document.getElementById("tNot").value;

  let updated;
  if (isEtkinlik) {
    const etkinlikDurum = document.getElementById("etkinlikDurum")?.value || "tamamlandi";
    // Durumu not alanına yaz, numerik alanlar sıfır kalır
    updated = {
      ...a,
      durum: "submitted",
      cozulen: 0, dogru: 0, yanlis: 0, bos: 0, net: 0,
      not: not || etkinlikDurum
    };
  } else {
    const coz = Number(document.getElementById("tCoz").value) || 0;
    const dog = Number(document.getElementById("tDog").value) || 0;
    const yan = Number(document.getElementById("tYan").value) || 0;
    const bos = Math.max(0, coz - dog - yan);
    if (dog + yan > coz) { alert("Doğru+Yanlış çözülenden fazla olamaz!"); return; }
    updated = {
      ...a,
      durum: "submitted",
      cozulen: coz, dogru: dog, yanlis: yan, bos,
      net: (dog - yan * 0.25).toFixed(2),
      not
    };
  }

  const tarih = document.getElementById("snvTarih").value;
  if (!tarih) { alert("Lütfen tarih seçin!"); return; }

function csvIndir(tip) {
  let rows = [];
  let dosyaAdi = "";
  if (tip === "odevler") {
    dosyaAdi = "odevler.csv";
    rows = [["id", "studentId", "ogrenciAd", "ders", "konu", "tarih", "verilen", "cozulen", "dogru", "yanlis", "net", "durum", "kocNot", "kaynakKitap", "sayfaNo"]];
    DB.odevler.forEach(a => {
      const ogr = DB.ogrenciler.find(o => o.id === a.studentId);
      rows.push([a.id, a.studentId, ogr ? ogr.ad : "", a.ders, a.konu, fmt(a.tarih), a.verilen, a.cozulen, a.dogru, a.yanlis, a.net, a.durum, a.kocNot || "", a.kaynakKitap || "", a.sayfaNo || ""]);
    });
  } else if (tip === "ogrenciler") {
    dosyaAdi = "ogrenciler.csv";
    rows = [["id", "ad", "kullaniciAdi", "sifre", "sinif", "renk"]];
    DB.ogrenciler.forEach(o => rows.push([o.id, o.ad, o.kullaniciAdi, o.sifre, o.sinif || "", o.renk || ""]));
  } else if (tip === "hedefler") {
    dosyaAdi = "hedefler.csv";
    rows = [["id", "ogrenciAd", "ders", "haftalikHedef"]];
    DB.hedefler.forEach(h => {
      const ogr = DB.ogrenciler.find(o => o.id === h.studentId);
      rows.push([h.id, ogr ? ogr.ad : "", h.ders, h.haftalikHedef]);
    });
  } else if (tip === "analiz") {
    dosyaAdi = "analiz_raporu.csv";
    rows = [["Öğrenci", "Ders", "Toplam Ödev", "Çözülen", "Doğru", "Yanlış", "Net"]];
    DB.ogrenciler.forEach(ogr => {
      const subMap = {};
      DB.odevler.filter(a => a.studentId === ogr.id).forEach(a => {
        if (!subMap[a.ders]) subMap[a.ders] = { count: 0, coz: 0, dog: 0, yan: 0 };
        subMap[a.ders].count++;
        subMap[a.ders].coz += a.cozulen;
        subMap[a.ders].dog += a.dogru;
        subMap[a.ders].yan += a.yanlis;
      });
      Object.entries(subMap).forEach(([ders, st]) => {
        rows.push([ogr.ad, ders, st.count, st.coz, st.dog, st.yan, (st.dog - st.yan * 0.25).toFixed(2)]);
      });
    });
  }
  const csv = rows.map(r => r.map(c => `"${String(c).replace(/"/g, '""')}"`).join(",")).join("\n");
  const blob = new Blob(["\uFEFF" + csv], { type: "text/csv;charset=utf-8;" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = dosyaAdi;
  a.click();
  URL.revokeObjectURL(url);
  toast("✅ İndiriliyor: " + dosyaAdi);
}

function csvAl(tip) {
  const input = document.createElement("input");
  input.type = "file";
  input.accept = ".csv";
  input.onchange = function(e) {
    const file = e.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = function(ev) {
      const text = ev.target.result.replace(/^\uFEFF/, "");
      const lines = text.split("\n").filter(l => l.trim());
      if (lines.length < 2) { alert("CSV en az 2 satır içermeli (başlık + veri)"); return; }
      const header = lines[0].split(",").map(h => h.trim().replace(/^"|"$/g, "").toLowerCase());
      let eklenen = 0;
      if (tip === "odevler") {
        for (let i = 1; i < lines.length; i++) {
          const cols = lines[i].split(",").map(c => c.trim().replace(/^"|"$/g, ""));
          const obj = {};
          header.forEach((h, idx) => obj[h] = cols[idx] || "");
          const yeni = {
            id: uid(),
            studentId: obj.studentid || obj.studentId || "",
            ders: obj.ders || "",
            konu: obj.konu || "",
            tarih: obj.tarih || bugun(),
            verilen: Number(obj.verilen) || 0,
            cozulen: 0,
            dogru: 0,
            yanlis: 0,
            bos: 0,
            net: 0,
            durum: "pending",
            not: "",
            kocNot: "",
            kocPuan: null,
            olusturma: bugun(),
            kaynakKitap: obj.kaynakkitap || obj.kaynakKitap || "",
            sayfaNo: obj.sayfano || obj.sayfaNo || ""
          };
          if (yeni.studentId && yeni.konu) {
            DB.odevler.push(parseOdev(yeni));
            eklenen++;
          }
        }
        toast(`✅ ${eklenen} ödev yüklendi`);
      } else if (tip === "ogrenciler") {
        for (let i = 1; i < lines.length; i++) {
          const cols = lines[i].split(",").map(c => c.trim().replace(/^"|"$/g, ""));
          const obj = {};
          header.forEach((h, idx) => obj[h] = cols[idx] || "");
          const yeni = {
            id: uid(),
            ad: obj.ad || "",
            kullaniciAdi: (obj.kullaniciadi || obj.kullanici_adi || "").toLowerCase(),
            sifre: obj.sifre || "1234",
            sinif: obj.sinif || "",
            renk: COLORS[DB.ogrenciler.length % COLORS.length],
            kocId: "coach1"
          };
          if (yeni.ad && yeni.kullaniciAdi) {
            DB.ogrenciler.push(yeni);
            eklenen++;
          }
        }
        toast(`✅ ${eklenen} öğrenci yüklendi`);
      }
      setTab("veri");
      if (!DEMO_MODE && SESSION && SESSION.role === "koc") verileriSenkronizeEt(false);
    };
    reader.readAsText(file, "UTF-8");
  };
  input.click();
}

function pdfAnalizCikar() {
  const { jsPDF } = window.jspdf;
  if (!jsPDF) { toast("❌ PDF kütüphanesi yüklenemedi", true); return; }

  const secilenDers = document.getElementById("pdfDersSecim")?.value || "__hepsi__";
  const ogrId = document.getElementById("analizOgr")?.value || "";
  const ogrAd = ogrId ? (DB.ogrenciler.find(o => o.id === ogrId)?.ad || "Tüm Öğrenciler") : "Tüm Öğrenciler";

  // Veriyi hesapla
  const odevler = DB.odevler.filter(a => !ogrId || a.studentId === ogrId);
  const subMap = {};
  const konuMap = {};
  odevler.forEach(a => {
    if (!subMap[a.ders]) subMap[a.ders] = { cozulen: 0, dogru: 0, yanlis: 0, count: 0 };
    subMap[a.ders].cozulen += a.cozulen || 0;
    subMap[a.ders].dogru   += a.dogru   || 0;
    subMap[a.ders].yanlis  += a.yanlis  || 0;
    subMap[a.ders].count++;
    if (!konuMap[a.ders]) konuMap[a.ders] = {};
    if (!konuMap[a.ders][a.konu]) konuMap[a.ders][a.konu] = { cozulen: 0, dogru: 0, yanlis: 0, count: 0 };
    konuMap[a.ders][a.konu].cozulen += a.cozulen || 0;
    konuMap[a.ders][a.konu].dogru   += a.dogru   || 0;
    konuMap[a.ders][a.konu].yanlis  += a.yanlis  || 0;
    konuMap[a.ders][a.konu].count++;
  });

  const dersListesi = secilenDers === "__hepsi__" ? Object.keys(subMap) : (subMap[secilenDers] ? [secilenDers] : []);
  if (dersListesi.length === 0) { toast("❌ Seçilen ders için veri yok", true); return; }

  const doc = new jsPDF({ orientation: "portrait", unit: "mm", format: "a4" });
  const PW = 210, ML = 14, MR = 14, CW = PW - ML - MR;
  let y = 0;

  function addPage() {
    doc.addPage();
    y = 16;
    // Üst şerit
    doc.setFillColor(26, 32, 53);
    doc.rect(0, 0, PW, 10, "F");
    doc.setFontSize(8);
    doc.setTextColor(167, 139, 250);
    doc.text("EduCoach — Analiz Raporu", ML, 7);
    doc.text(bugun(), PW - MR, 7, { align: "right" });
  }

  function checkY(needed) {
    if (y + needed > 277) addPage();
  }

  // ── KAPAK ──────────────────────────────────────────────────
  // Arkaplan
  doc.setFillColor(13, 17, 23);
  doc.rect(0, 0, PW, 297, "F");

  // Üst bant
  doc.setFillColor(124, 58, 237);
  doc.rect(0, 0, PW, 42, "F");

  // Başlık
  doc.setFontSize(22);
  doc.setTextColor(255, 255, 255);
  doc.setFont("helvetica", "bold");
  doc.text("EduCoach", ML, 18);
  doc.setFontSize(12);
  doc.setTextColor(196, 181, 253);
  doc.text("Analiz Raporu — Derse Gore Detay", ML, 27);
  doc.setFontSize(9);
  doc.setTextColor(255, 255, 255);
  doc.text("Ogrenci: " + ogrAd, ML, 37);

  // Tarih
  doc.setFontSize(9);
  doc.setTextColor(100, 116, 139);
  doc.text("Olusturulma: " + bugun(), PW - MR, 37, { align: "right" });

  // Genel özet kutu
  const totCoz = odevler.reduce((s,a) => s + (a.cozulen||0), 0);
  const totD   = odevler.reduce((s,a) => s + (a.dogru||0),   0);
  const totY   = odevler.reduce((s,a) => s + (a.yanlis||0),  0);
  const totNet = (totD - totY * 0.25).toFixed(2);

  doc.setFillColor(26, 32, 53);
  doc.roundedRect(ML, 52, CW, 32, 4, 4, "F");
  doc.setFontSize(9);
  doc.setTextColor(148, 163, 184);
  doc.text("Genel Ozet", ML + 6, 60);

  const cols = [
    ["Toplam Odev", odevler.length],
    ["Cozulen Soru", totCoz],
    ["Dogru", totD],
    ["Yanlis", totY],
    ["Toplam Net", totNet]
  ];
  const colW = CW / cols.length;
  cols.forEach(([lbl, val], i) => {
    const cx = ML + i * colW + colW / 2;
    doc.setFontSize(14);
    doc.setTextColor(167, 139, 250);
    doc.setFont("helvetica", "bold");
    doc.text(String(val), cx, 73, { align: "center" });
    doc.setFontSize(8);
    doc.setTextColor(100, 116, 139);
    doc.setFont("helvetica", "normal");
    doc.text(lbl, cx, 79, { align: "center" });
  });

  y = 95;

  // ── DERS BLOKLARI ──────────────────────────────────────────
  dersListesi.forEach((ders, di) => {
    const st = subMap[ders];
    const konular = konuMap[ders] || {};
    const pct = st.cozulen > 0 ? Math.min(100, Math.round(st.dogru / st.cozulen * 100)) : 0;
    const net = (st.dogru - st.yanlis * 0.25).toFixed(2);

    // Renk eşleşmesi (SC sözlüğünden)
    const hexToRgb = hex => {
      const r = parseInt(hex.slice(1,3),16),
            g = parseInt(hex.slice(3,5),16),
            b = parseInt(hex.slice(5,7),16);
      return [r,g,b];
    };
    const dersHex = SC[ders] || "#7c3aed";
    const [dr,dg,db] = hexToRgb(dersHex);

    // Yeni sayfaya geç mi?
    checkY(48);

    // ─ Ders başlık bant ─
    doc.setFillColor(26, 32, 53);
    doc.roundedRect(ML, y, CW, 36, 4, 4, "F");
    // Sol renkli çizgi
    doc.setFillColor(dr, dg, db);
    doc.rect(ML, y, 4, 36, "F");

    doc.setFontSize(12);
    doc.setTextColor(dr, dg, db);
    doc.setFont("helvetica", "bold");
    doc.text(ders, ML + 8, y + 8);

    // Doğruluk progress bar (metin)
    doc.setFontSize(9);
    doc.setTextColor(148, 163, 184);
    doc.setFont("helvetica", "normal");
    doc.text("Dogruluk: " + pct + "%", ML + 8, y + 16);
    doc.text("Cozulen: " + st.cozulen + "   D: " + st.dogru + "   Y: " + st.yanlis + "   Odev: " + st.count, ML + 8, y + 23);
    doc.setTextColor(167, 139, 250);
    doc.setFontSize(10);
    doc.setFont("helvetica", "bold");
    doc.text("Net: " + net, PW - MR - 2, y + 14, { align: "right" });

    // Progress bar grafik
    const barX = ML + 8, barY = y + 27, barW = CW - 20, barH = 4;
    doc.setFillColor(30, 41, 59);
    doc.roundedRect(barX, barY, barW, barH, 2, 2, "F");
    const fillColor = pct >= 70 ? [34,197,94] : pct >= 50 ? [245,158,11] : [248,113,113];
    doc.setFillColor(...fillColor);
    doc.roundedRect(barX, barY, Math.max(2, barW * pct / 100), barH, 2, 2, "F");

    y += 42;

    // ─ Konu satırları ─
    const konuSirali = Object.entries(konular).sort((a, b) => {
      const pA = a[1].cozulen > 0 ? a[1].dogru / a[1].cozulen : 0;
      const pB = b[1].cozulen > 0 ? b[1].dogru / b[1].cozulen : 0;
      return pA - pB; // zayıftan güçlüye
    });

    if (konuSirali.length > 0) {
      checkY(10);
      doc.setFontSize(8);
      doc.setTextColor(71, 85, 105);
      doc.setFont("helvetica", "bold");
      doc.text("KONULAR (ZAYIFTAN GUCLUE)", ML + 4, y);
      y += 5;

      konuSirali.forEach(([konu, ks]) => {
        checkY(14);
        const kpct = ks.cozulen > 0 ? Math.min(100, Math.round(ks.dogru / ks.cozulen * 100)) : 0;
        const knet = (ks.dogru - ks.yanlis * 0.25).toFixed(2);
        const kColor = kpct >= 70 ? [34,197,94] : kpct >= 50 ? [245,158,11] : [248,113,113];

        // Sol ince bant
        doc.setFillColor(...kColor);
        doc.rect(ML + 4, y, 2, 10, "F");

        doc.setFillColor(19, 25, 41);
        doc.rect(ML + 7, y, CW - 7, 10, "F");

        // Konu adı (kısalt)
        const konuKisa = konu.length > 38 ? konu.slice(0, 36) + "…" : konu;
        doc.setFontSize(8);
        doc.setTextColor(226, 232, 240);
        doc.setFont("helvetica", "normal");
        doc.text(konuKisa, ML + 11, y + 6.5);

        // Sağ taraf değerler
        doc.setTextColor(...kColor);
        doc.setFont("helvetica", "bold");
        doc.text(kpct + "%", PW - MR - 28, y + 6.5, { align: "right" });
        doc.setTextColor(167, 139, 250);
        doc.text("Net:" + knet, PW - MR - 14, y + 6.5, { align: "right" });
        doc.setTextColor(71, 85, 105);
        doc.setFont("helvetica", "normal");
        doc.text(ks.count + " od.", PW - MR, y + 6.5, { align: "right" });

        y += 11;
      });
    }

    y += 6; // ders arası boşluk
  });

  // ── FOOTER ─────────────────────────────────────────────────
  const pageCount = doc.internal.getNumberOfPages();
  for (let i = 1; i <= pageCount; i++) {
    doc.setPage(i);
    doc.setFontSize(8);
    doc.setTextColor(51, 65, 85);
    doc.text("EduCoach | " + bugun(), ML, 292);
    doc.text(i + " / " + pageCount, PW - MR, 292, { align: "right" });
  }

  // Dosya adı
  const dosyaAd = "EduCoach_Analiz_" + (secilenDers === "__hepsi__" ? "TumDersler" : secilenDers.replace(/\s/g,"_")) + "_" + (ogrId ? ogrAd.replace(/\s/g,"_") : "GenelRapor") + "_" + bugun() + ".pdf";
  doc.save(dosyaAd);
  toast("✅ PDF oluşturuldu ve indiriliyor...");
}

function _filterOgrImpl() {
  const fDurum = document.getElementById("fOgrDurum")?.value || "";
  const fDers = document.getElementById("fOgrDers")?.value || "";
  const fTarihBas = document.getElementById("fOgrTarihBas")?.value || "";
  const fTarihBit = document.getElementById("fOgrTarihBit")?.value || "";
  const gövde = document.getElementById("ogrOdevListesiGövde");
  if (!gövde) return;

  let liste = DB.odevler.filter(a => a.studentId === SESSION.id);

  // Filtreleri Uygula
  if (fDers) liste = liste.filter(a => a.ders === fDers);
  
  if (fDurum === "gecmis") {
    liste = liste.filter(a => a.durum === "pending" && gecmis(fmt(a.tarih)));
  } else if (fDurum) {
    liste = liste.filter(a => a.durum === fDurum);
  }

  if (fTarihBas) liste = liste.filter(a => fmt(a.tarih) >= fTarihBas);
  if (fTarihBit) liste = liste.filter(a => fmt(a.tarih) <= fTarihBit);

  // Tarihe göre sırala (Yeni en üstte)
  liste.sort((a, b) => String(b.tarih).localeCompare(String(a.tarih)));

  if (liste.length === 0) {
    gövde.innerHTML = `<div style="text-align:center; padding:40px; color:#475569">Aranan kriterde ödev bulunamadı.</div>`;
    return;
  }

  gövde.innerHTML = liste.map(a => {
    const isOverdue = a.durum === "pending" && gecmis(fmt(a.tarih));
    const canSubmit = a.durum === "pending";
    const canRetract = a.durum === "submitted";
    return ogrOdevKarti(a, canSubmit, canRetract);
  }).join("");
}
